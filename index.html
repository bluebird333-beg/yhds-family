const functions = require("firebase-functions/v2"); const admin = require("firebase-admin"); admin.initializeApp(); const db = admin.firestore(); const messaging = admin.messaging(); // 토큰 가져오기 (사용자별 최신 토큰 1개만) async function getTokens(excludeUser = null, onlyUsers = null) { const tokensSnap = await db.collection("fcmTokens").get(); const userTokenMap = {}; // user -> 최신 토큰 1개만 tokensSnap.forEach(doc => { const data = doc.data(); if (!data.token || !data.user) return; if (excludeUser && data.user === excludeUser) return; if (onlyUsers && !onlyUsers.has(data.user)) return; // 같은 유저면 최신 토큰으로 덮어쓰기 if (!userTokenMap[data.user]) { userTokenMap[data.user] = data.token; } }); return Object.values(userTokenMap); } // 새 글 알림 exports.onNewPost = functions.firestore.onDocumentCreated("posts/{postId}", async (event) => { const post = event.data.data(); if (!post) return; const author = post.author || "누군가"; const content = (post.content || "새 글을 확인해보세요!").slice(0, 60); const category = post.category || ""; const tokens = await getTokens(author); if (!tokens.length) return; try { await messaging.sendEachForMulticast({ notification: { title: `🏠 ${author}이(가) [${category}]에 글을 올렸어요!`, body: content }, tokens }); } catch (e) { console.error("글 알림 오류:", e); } }); // 새 댓글 알림 exports.onNewComment = functions.firestore.onDocumentUpdated("posts/{postId}", async (event) => { const before = event.data.before.data(); const after = event.data.after.data(); if (!before || !after) return; const beforeCmts = before.comments || []; const afterCmts = after.comments || []; if (afterCmts.length <= beforeCmts.length) return; const newComment = afterCmts[afterCmts.length - 1]; const commenter = newComment.author || "누군가"; const text = (newComment.text || "댓글을 확인해보세요!").slice(0, 60); const postAuthor = after.author || ""; const notifyUsers = new Set(); notifyUsers.add(postAuthor); beforeCmts.forEach(c => { if (c.author) notifyUsers.add(c.author); }); notifyUsers.delete(commenter); if (!notifyUsers.size) return; const tokens = await getTokens(commenter, notifyUsers); if (!tokens.length) return; try { await messaging.sendEachForMulticast({ notification: { title: `💬 ${commenter}이(가) 댓글을 달았어요!`, body: `"${text}"` }, tokens }); } catch (e) { console.error("댓글 알림 오류:", e); } });
